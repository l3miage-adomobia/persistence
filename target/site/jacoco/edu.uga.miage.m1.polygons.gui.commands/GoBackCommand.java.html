    private JDrawingFrame jDrawingFrame;
    public GoBackCommand(JDrawingFrame jDrawingFrame){
        this.jDrawingFrame = jDrawingFrame;
    }
    @Override
    public void execute() {
        jDrawingFrame.goBack();
